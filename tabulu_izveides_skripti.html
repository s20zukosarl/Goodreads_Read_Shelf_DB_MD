Tabula "Autori" 

CREATE TABLE `autori` (
  `idautori` int(11) NOT NULL AUTO_INCREMENT,
  `uzvārds` varchar(65) DEFAULT NULL,
  `vārds` varchar(65) DEFAULT NULL,
  `idvalstis` int(11) NOT NULL,
  PRIMARY KEY (`idautori`),
  KEY `idvalstis_idx` (`idvalstis`),
  CONSTRAINT `idvalstis` FOREIGN KEY (`idvalstis`) REFERENCES `valstis` (`idvalstis`) ON DELETE NO ACTION ON UPDATE CASCADE
) ENGINE=InnoDB AUTO_INCREMENT=54 DEFAULT CHARSET=utf8;

Tabula "Grāmatas" 

CREATE TABLE `grāmatas` (
  `idgrāmatas` int(11) NOT NULL AUTO_INCREMENT,
  `nosaukums` varchar(255) NOT NULL,
  `idautori` int(11) NOT NULL,
  `idžanrs` int(11) NOT NULL,
  `lpp_skaits` int(11) DEFAULT NULL,
  `izdošanas_gads` int(11) DEFAULT NULL,
  `oriģ_izd_gads` int(11) DEFAULT NULL,
  `idreitings` int(11) NOT NULL,
  `idizdevniecības` int(11) NOT NULL,
  `idlasīšanas_gads` int(11) NOT NULL,
  PRIMARY KEY (`idgrāmatas`),
  KEY `idautori_idx` (`idautori`),
  KEY `idžanrs_idx` (`idžanrs`),
  KEY `idreitings_idx` (`idreitings`),
  KEY `idizdevniecības_idx` (`idizdevniecības`),
  KEY `idlasīšanas_gads_idx` (`idlasīšanas_gads`),
  CONSTRAINT `idautori` FOREIGN KEY (`idautori`) REFERENCES `autori` (`idautori`) ON DELETE NO ACTION ON UPDATE CASCADE,
  CONSTRAINT `idizdevniecības` FOREIGN KEY (`idizdevniecības`) REFERENCES `izdevniecības` (`idizdevniecības`) ON DELETE NO ACTION ON UPDATE CASCADE,
  CONSTRAINT `idlasīšanas_gads` FOREIGN KEY (`idlasīšanas_gads`) REFERENCES `lasīšanasgads` (`idlasīšanas_gads`) ON DELETE NO ACTION ON UPDATE CASCADE,
  CONSTRAINT `idreitings` FOREIGN KEY (`idreitings`) REFERENCES `reitings` (`idreitings`) ON DELETE NO ACTION ON UPDATE CASCADE,
  CONSTRAINT `idžanrs` FOREIGN KEY (`idžanrs`) REFERENCES `žanrs` (`idžanrs`) ON DELETE NO ACTION ON UPDATE CASCADE
) ENGINE=InnoDB AUTO_INCREMENT=254 DEFAULT CHARSET=utf8;
